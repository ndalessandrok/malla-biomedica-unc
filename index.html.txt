        function checkPrerequisites(subject) {
            if (!subject.prerequisites || subject.prerequisites.length === 0) {
                return true;
            }
            
            return subject.prerequisites.every(prereq => {
                const prereqSubject = subjects[prereq];
                return prereqSubject && prereqSubject.completed;
            });
        }

        function getSubjectStatus(subject) {
            if (subject.completed) {
                return 'completed';
            }
            
            if (checkPrerequisites(subject)) {
                return 'available';
            }
            
            return 'locked';
        }

        function toggleSubject(subjectId) {
            const subject = subjects[subjectId];
            if (!subject) return;

            const currentStatus = getSubjectStatus(subject);
            
            if (currentStatus === 'locked') {
                showNotification('‚ö†Ô∏è Esta materia no est√° disponible. Completa las correlativas primero.');
                return;
            }

            // Toggle completion status
            subject.completed = !subject.completed;
            
            if (subject.completed) {
                // Prompt for grade
                const grade = prompt('Ingresa tu nota (opcional):');
                if (grade) {
                    subject.grade = grade;
                }
                
                // Check for newly available subjects
                const newlyAvailable = [];
                Object.keys(subjects).forEach(id => {
                    const s = subjects[id];
                    if (!s.completed && checkPrerequisites(s)) {
                        if (s.prerequisites.includes(subjectId)) {
                            newlyAvailable.push(s.name);
                        }
                    }
                });
                
                if (newlyAvailable.length > 0) {
                    showNotification(`üéâ ¬°Nuevas materias disponibles! ${newlyAvailable.join(', ')}`);
                } else {
                    showNotification('‚úÖ Materia completada');
                }
            } else {
                subject.grade = null;
                showNotification('‚ùå Materia desmarcada');
            }
            
            renderMalla();
        }

        function showNotification(message) {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.classList.add('show');
            
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }

            });

            // Render each semester

                if (!subjectsBySemester[semester]) return;

                const semesterDiv = document.createElement('div');
                semesterDiv.className = 'semester';
                
                const semesterNumber = semesterOrder.indexOf(semester);
                const semesterIcon = semesterNumber === 0 ? 'N' : semesterNumber;
                
                semesterDiv.innerHTML = `

                    <div class="subjects" id="subjects-${semester}"></div>
                `;

        function updateStats() {
            const totalSubjects = Object.keys(subjects).length;
            const completedSubjects = Object.values(subjects).filter(s => s.completed).length;
            const availableSubjects = Object.values(subjects).filter(s => !s.completed && checkPrerequisites(s)).length;
            const totalCredits = Object.values(subjects).filter(s => s.completed).reduce((sum, s) => sum + s.credits, 0);
            
            const progressPercentage = Math.round((completedSubjects / totalSubjects) * 100);
            
            document.getElementById('completed-count').textContent = completedSubjects;
            document.getElementById('available-count').textContent = availableSubjects;
            document.getElementById('total-count').textContent = totalSubjects;
            document.getElementById('credits-count').textContent = totalCredits;
            
            const progressFill = document.getElementById('progress-fill');
            progressFill.style.width = `${progressPercentage}%`;
            progressFill.textContent = `${progressPercentage}%`;
        }
